
fn render_series_episodes_pane(f: &mut Frame, app: &mut App, area: Rect, border_color: Color) {
    let title = format!(" // EPISODES ({}) ", app.series_episodes.len());

    let block = Block::default()
        .borders(Borders::ALL)
        .border_style(Style::default().fg(if app.active_pane == Pane::Episodes {
            BRIGHT_GREEN
        } else {
            border_color
        }))
        .border_type(BorderType::Double)
        .title(Span::styled(
            title,
            Style::default()
                .fg(MATRIX_GREEN)
                .add_modifier(Modifier::BOLD),
        ));

    let items: Vec<ListItem> = app
        .series_episodes
        .iter()
        .map(|ep| {
            let style = if app.active_pane == Pane::Episodes {
                Style::default().fg(MATRIX_GREEN)
            } else {
                Style::default().fg(DARK_GREEN)
            };

            let episode_title = ep.title.as_deref().unwrap_or("Untitled");
            let display = format!(
                "S{:02}E{:02} - {}",
                ep.season, ep.episode_num, episode_title
            );

            ListItem::new(Line::from(vec![
                Span::styled(" [ EP ] ", style),
                Span::styled(display, style),
            ]))
        })
        .collect();

    let list = List::new(items)
        .block(block)
        .highlight_style(
            Style::default()
                .bg(MATRIX_GREEN)
                .fg(Color::Black)
                .add_modifier(Modifier::BOLD),
        )
        .highlight_symbol(" Â» ");

    f.render_stateful_widget(list, area, &mut app.series_episode_list_state);
}
